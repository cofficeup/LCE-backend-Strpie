<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;

/**
 * Legacy user promocodes table
 * Note: This is the old promocode table, newer implementation uses UserPromoCode model
 */
class UserPromocodes extends Model
{
    protected $table = 'lce_user_promocodes';
    public $timestamps = false;

    protected $fillable = [
        'user_id',
        'promocode_id',
        'promocode_name',
        'used_date',
        'cdate',
    ];

    /**
     * Get the user associated with this promo code usage
     */
    public function user()
    {
        return $this->belongsTo(User::class, 'user_id');
    }

    /**
     * Get the promo code
     */
    public function promoCode()
    {
        return $this->belongsTo(PromoCode::class, 'promocode_id');
    }
}
